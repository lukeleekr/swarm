def fib(n: int) -> int:
    if n < 0:
        raise ValueError('n must be non-negative')

    if n < 2:
        return n

    prev = 0
    curr = 1
    for _ in range(2, n + 1):
        prev, curr = curr, prev + curr
    return curr
